110    def get_geotiff_extent(self, fname: str):
111        """Get info from GeoTIFF on its extent
112
113        Args:
114            fname (str): path of GeoTIFF file
115
116        Raises:
117            ValueError: _description_
118            IOError: _description_
119
120        Returns:
121            tuple(int,int,int,int,int,int,int): width,height,top_left,top_right,bottom_left,
122            bottom_right,pixel_width
123        """
124        try:
125            with rasterio.open(fname) as dataset:
126                transform = dataset.transform
127                width = dataset.width
128                height = dataset.height
129
130                top_left = transform * (0, 0)
131                top_right = transform * (width, 0)
132                bottom_left = transform * (0, height)
133                bottom_right = transform * (width, height)
134
135                pixel_width = transform[0]
136                pixel_height = -transform[4]  # Negative because the height is
137                # typically negative in GeoTIFFs
138                if int(pixel_width) != int(pixel_height):
139                    raise ValueError(f"pixel_width {pixel_width} != pixel_height {pixel_height}")
140        except RasterioIOError as exc:
141            raise IOError(f"Could not read GeoTIFF: {exc}") from exc
142        return (
143            width,
144            height,
145            top_left,
146            top_right,
147            bottom_left,
148            bottom_right,
149            pixel_width,
150        )

480    def get_segment(
481        self, segment_bounds: list, grid_xy: bool = True, flatten: bool = False
482    ) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
483        """return a cropped segment of the slopes, flattened or as a grid
484
485        Args:
486            segment_bounds (List): [(minx,maxx),(miny,maxy)]
487            grid_xy (bool, optional): return segment as a grid. Defaults to True.
488            flatten (bool, optional): return segment as flattened list. Defaults to False.
489
490        Returns:
491            Tuple[np.ndarray, np.ndarray, np.ndarray]: (xslopes,yslopes,zslopes)
492        """
493
494        # ----------------------------------------------------------------------
495        # Get coord bounds as index bounds
496        # ----------------------------------------------------------------------
497
498        minx_ind = (np.absolute(segment_bounds[0][0] - self.xslopes)).argmin()
499        maxx_ind = (np.absolute(segment_bounds[0][1] - self.xslopes)).argmin()
500        miny_ind = (np.absolute(segment_bounds[1][0] - self.yslopes)).argmin()
501        maxy_ind = (np.absolute(segment_bounds[1][1] - self.yslopes)).argmin()
502
503        # ----------------------------------------------------------------------
504        # Crop full slopes coords to segment bounds
505        # ----------------------------------------------------------------------
506
507        zslopes = self.zslopes[maxy_ind:miny_ind, minx_ind:maxx_ind]
508        xslopes = self.xslopes[minx_ind:maxx_ind]
509        yslopes = self.yslopes[maxy_ind:miny_ind]
510
511        if grid_xy is True:
512            xslopes, yslopes = np.meshgrid(xslopes, yslopes)
513
514            # Set x,y to nan where z is nan
515            zslopes_nan = np.isnan(zslopes)
516            xslopes[zslopes_nan] = np.nan
517            yslopes[zslopes_nan] = np.nan
518
519        # ----------------------------------------------------------------------
520        # Return, flattened if requested
521        # ----------------------------------------------------------------------
522
523        if flatten is False:
524            return (xslopes, yslopes, zslopes)
525
526        return (xslopes.flatten(), yslopes.flatten(), zslopes.flatten())

528    def chunked_interpolation(
529        self, x: np.ndarray, y: np.ndarray, myyslopes: np.ndarray, xslopes: np.ndarray, method: str
530    ) -> np.ndarray:
531        """Interpolate slopes in chunks to handle large datasets efficiently.
532
533        This function performs interpolation on a slopes stored in a Zarr array by
534        extracting relevant chunks and creating a sub-grid for interpolation.
535
536        Args:
537            x (np.ndarray): Array of x coordinates in the slopes's projection (in meters).
538            y (np.ndarray): Array of y coordinates in the slopes's projection (in meters).
539            myyslopes (np.ndarray): Flipped y coordinates corresponding to the slopes grid.
540            xslopes (np.ndarray): x coordinates corresponding to the slopes grid.
541            method (str): Interpolation method to use ('linear', 'nearest', etc.).
542
543        Returns:
544            np.ndarray: Interpolated slopes elevation values at the specified coordinates.
545        """
546        results = np.full_like(x, np.nan, dtype=np.float64)
547
548        # Identify valid points (where x and y are not NaN)
549        valid_mask = ~np.isnan(x) & ~np.isnan(y)
550
551        # Only proceed if there are valid points
552        if valid_mask.any():
553            x_valid = x[valid_mask]
554            y_valid = y[valid_mask]
555
556            # Define the bounding box for valid points
557            x_min, x_max = x_valid.min(), x_valid.max()
558            y_min, y_max = y_valid.min(), y_valid.max()
559
560            # Determine the indices of the bounding box in the slopes grid
561            x_indices = np.searchsorted(xslopes, [x_min, x_max])
562            y_indices = np.searchsorted(myyslopes, [y_min, y_max])
563
564            # Expand the indices to ensure we cover the region adequately
565            x_indices[0] = max(x_indices[0] - 1, 0)
566            x_indices[1] = min(x_indices[1] + 1, len(xslopes) - 1)
567            y_indices[0] = max(y_indices[0] - 1, 0)
568            y_indices[1] = min(y_indices[1] + 1, len(myyslopes) - 1)
569
570            # Extract the sub-array
571            sub_zarr = self.zslopes_flip[
572                y_indices[0] : y_indices[1] + 1, x_indices[0] : x_indices[1] + 1
573            ]
574            sub_zarr = np.array(sub_zarr)
575
576            sub_myyslopes = myyslopes[y_indices[0] : y_indices[1] + 1]
577            sub_xslopes = xslopes[x_indices[0] : x_indices[1] + 1]
578
579            # Create an interpolator for the sub-array
580            interpolator = RegularGridInterpolator(
581                (sub_myyslopes, sub_xslopes),
582                sub_zarr,
583                method=method,
584                bounds_error=False,
585                fill_value=np.nan,
586            )
587

589            points = np.vstack((y_valid, x_valid)).T
590            interpolated_values = interpolator(points)
591
592            # Store the results in the corresponding places
593            results[valid_mask] = interpolated_values

623
624        x = np.array(x)
625        y = np.array(y)

628        if xy_is_latlon:
629            x, y = self.lonlat_to_xy_transformer.transform(  # pylint: disable=E0633
630                y, x

641            (myyslopes, self.xslopes),

646            fill_value=np.nan,

651        perform a gaussian smooth on the current loaded slopes
652        sigma : degree of smoothing, def=1.0
653        """
654        # Gaussian smooth slopes
655        this_zslopes = self.zslopes.copy()

663    def hillshade(self, azimuth=225, pitch=45):
664        """
665        Convert the slopes 'z_slopes' values to a hillshade value between 0..255
666        azimuth: angle in degrees (0..360)
667        pitch : angle in degrees (0..90)
668        """
669        azimuth = 360.0 - azimuth
670
671        x, y = np.gradient(self.zslopes)
672        slope = np.pi / 2.0 - np.arctan(np.sqrt(x * x + y * y))
673        aspect = np.arctan2(-x, y)
674        azimuthrad = azimuth * np.pi / 180.0
675        altituderad = pitch * np.pi / 180.0
676
677        shaded = np.sin(altituderad) * np.sin(slope) + np.cos(altituderad) * np.cos(slope) * np.cos(
678            (azimuthrad - np.pi / 2.0) - aspect
679        )
680
681        self.zslopes = 255 * (shaded + 1) / 2

529    def chunked_interpolation(
530        self, x: np.ndarray, y: np.ndarray, myyslopes: np.ndarray, xslopes: np.ndarray, method: str
531    ) -> np.ndarray:
532        """Interpolate slopes in chunks to handle large datasets efficiently.
533
534        This function performs interpolation on a slopes stored in a Zarr array by
535        extracting relevant chunks and creating a sub-grid for interpolation.
536
537        Args:
538            x (np.ndarray): Array of x coordinates in the slopes's projection (in meters).
539            y (np.ndarray): Array of y coordinates in the slopes's projection (in meters).
540            myyslopes (np.ndarray): Flipped y coordinates corresponding to the slopes grid.
541            xslopes (np.ndarray): x coordinates corresponding to the slopes grid.
542            method (str): Interpolation method to use ('linear', 'nearest', etc.).
543
544        Returns:
545            np.ndarray: Interpolated slopes elevation values at the specified coordinates.
546        """
547        results = np.full_like(x, np.nan, dtype=np.float64)
548
549        # Identify valid points (where x and y are not NaN)
550        valid_mask = ~np.isnan(x) & ~np.isnan(y)
551
552        # Only proceed if there are valid points
553        if valid_mask.any():
554            x_valid = x[valid_mask]
555            y_valid = y[valid_mask]
556
557            # Define the bounding box for valid points
558            x_min, x_max = x_valid.min(), x_valid.max()
559            y_min, y_max = y_valid.min(), y_valid.max()
560
561            # Determine the indices of the bounding box in the slopes grid
562            x_indices = np.searchsorted(xslopes, [x_min, x_max])
563            y_indices = np.searchsorted(myyslopes, [y_min, y_max])
564
565            # Expand the indices to ensure we cover the region adequately
566            x_indices[0] = max(x_indices[0] - 1, 0)
567            x_indices[1] = min(x_indices[1] + 1, len(xslopes) - 1)
568            y_indices[0] = max(y_indices[0] - 1, 0)
569            y_indices[1] = min(y_indices[1] + 1, len(myyslopes) - 1)
570
571            # Extract the sub-array
572            sub_zarr = self.zslopes_flip[
573                y_indices[0] : y_indices[1] + 1, x_indices[0] : x_indices[1] + 1
574            ]
575            sub_zarr = np.array(sub_zarr)
576
577            sub_myyslopes = myyslopes[y_indices[0] : y_indices[1] + 1]
578            sub_xslopes = xslopes[x_indices[0] : x_indices[1] + 1]
579
580            # Create an interpolator for the sub-array
581            interpolator = RegularGridInterpolator(
582                (sub_myyslopes, sub_xslopes),
583                sub_zarr,
584                method=method,
585                bounds_error=False,
586                fill_value=np.nan,
587            )
588
589            # Perform the interpolation for valid points
590            points = np.vstack((y_valid, x_valid)).T
591            interpolated_values = interpolator(points)
592
593            # Store the results in the corresponding places
594            results[valid_mask] = interpolated_values
595
596        return results

680        )
681
