708        self,
709        lats: np.ndarray | list,
710        lons: np.ndarray | list,
711        basin_numbers: Optional[list[int]] = None,
712        inputs_are_xy: bool = False,
713    ) -> tuple[np.ndarray, int]:
714        """Given a list of lat,lon or x,y points, find the points that are inside the current mask
715
716        Args:
717            lats (np.ndarray|list[float]): list of latitude points
718            lons (np.ndarray|list[float]): list of longitude points
719            basin_numbers (list[int,], optional): list of basin numbers. Defaults to None.
720            inputs_are_xy (bool, optional): lats, lons are already transformed to x,y.
721                                            Defaults to False.
722
723        Returns:
724            inmask(np.ndarray) : boolean array same size as input list, indicating whether
725            inputs points are inside (True) or outside (False) mask
726            n_inside (int) : number inside mask
727        """
728
729        if not self.mask_name:
730            inmask = np.zeros(len(lats), np.bool_)
731            return inmask, 0
732
733        if not isinstance(lats, np.ndarray):
734            if isinstance(lats, list):
735                lats = np.array(lats)
736            else:
737                raise TypeError("lats is wrong type. Must be np.ndarray or list[float]")
738
739        if not isinstance(lons, np.ndarray):
740            if isinstance(lons, list):
741                lons = np.array(lons)
742            else:
743                raise TypeError("lons is wrong type. Must be np.ndarray or list[float]")
744
745        if basin_numbers:  # turn in to a list if a scalar
746            if not isinstance(basin_numbers, (list, np.ndarray)):
747                basin_numbers = [basin_numbers]
748
749        if self.basin_numbers:
750            if not basin_numbers:
751                basin_numbers = self.basin_numbers
752
753        if inputs_are_xy:
754            x, y = lats, lons
755        else:
756            x, y = self.latlon_to_xy(lats, lons)  # pylint: disable=E0633
757
758        inmask = np.zeros(lats.size, np.bool_)
759
760        n_inside = 0
761        # ---------------------------------------------------------
762        # Find points inside a x,y rectangular limits mask
763        # ---------------------------------------------------------
764
765        if self.mask_type == "xylimits":
766            for i in range(x.size):
767                if (x[i] >= self.xlimits[0] and x[i] <= self.xlimits[1]) and (
768                    y[i] >= self.ylimits[0] and y[i] <= self.ylimits[1]
769                ):
770                    inmask[i] = True
771                    n_inside += 1
772            return inmask, n_inside
773
774        if self.mask_type == "grid":

776                # calculate equivalent (ii,jj) in mask array
777                ii = int(np.around((x[i] - self.minxm) / self.binsize))
778                jj = int(np.around((y[i] - self.minym) / self.binsize))

789                            inmask[i] = True
790                            n_inside += 1
791                else:
792                    if self.mask_grid[jj, ii] > 0:
793                        inmask[i] = True
794                        n_inside += 1
795        else:
796            return inmask, 0
797
798        return inmask, n_inside

708    def points_inside(
709        self,
710        lats: np.ndarray | list,
711        lons: np.ndarray | list,
712        basin_numbers: Optional[list[int]] = None,
713        inputs_are_xy: bool = False,
714    ) -> tuple[np.ndarray, int]:
715        """Given a list of lat,lon or x,y points, find the points that are inside the current mask
716
717        Args:
718            lats (np.ndarray|list[float]): list of latitude points
719            lons (np.ndarray|list[float]): list of longitude points
720            basin_numbers (list[int,], optional): list of basin numbers. Defaults to None.
721            inputs_are_xy (bool, optional): lats, lons are already transformed to x,y.
722                                            Defaults to False.
723
724        Returns:
725            inmask(np.ndarray) : boolean array same size as input list, indicating whether
726            inputs points are inside (True) or outside (False) mask
727            n_inside (int) : number inside mask
728        """
729
730        if not self.mask_name:
731            inmask = np.zeros(len(lats), np.bool_)
732            return inmask, 0
733
734        if not isinstance(lats, np.ndarray):
735            if isinstance(lats, list):
736                lats = np.array(lats)
737            else:
738                raise TypeError("lats is wrong type. Must be np.ndarray or list[float]")
739
740        if not isinstance(lons, np.ndarray):
741            if isinstance(lons, list):
742                lons = np.array(lons)
743            else:
744                raise TypeError("lons is wrong type. Must be np.ndarray or list[float]")
745
746        if basin_numbers:  # turn in to a list if a scalar
747            if not isinstance(basin_numbers, (list, np.ndarray)):
748                basin_numbers = [basin_numbers]
749
750        if self.basin_numbers:
751            if not basin_numbers:
752                basin_numbers = self.basin_numbers
753
754        if inputs_are_xy:
755            x, y = lats, lons
756        else:
757            x, y = self.latlon_to_xy(lats, lons)  # pylint: disable=E0633
758
759        inmask = np.zeros(lats.size, np.bool_)
760
761        n_inside = 0
762        # ---------------------------------------------------------
763        # Find points inside a x,y rectangular limits mask
764        # ---------------------------------------------------------
765
766        if self.mask_type == "xylimits":
767            for i in range(x.size):
768                if (x[i] >= self.xlimits[0] and x[i] <= self.xlimits[1]) and (
769                    y[i] >= self.ylimits[0] and y[i] <= self.ylimits[1]
770                ):
771                    inmask[i] = True
772                    n_inside += 1
773            return inmask, n_inside
774
775        if self.mask_type == "grid":
776            for i in range(x.size):
777                # calculate equivalent (ii,jj) in mask array
778                ii = int(np.around((x[i] - self.minxm) / self.binsize))
779                jj = int(np.around((y[i] - self.minym) / self.binsize))
780
781                # Check bounds of Basin Mask array
782                if ii < 0 or ii >= self.num_x:
783                    continue
784                if jj < 0 or jj >= self.num_y:
785                    continue
786
787                if basin_numbers:
788                    for basin in basin_numbers:
789                        if self.mask_grid[jj, ii] == basin:
790                            inmask[i] = True
791                            n_inside += 1
792                else:
793                    if self.mask_grid[jj, ii] > 0:
794                        inmask[i] = True
795                        n_inside += 1
796        else:
797            return inmask, 0
798
799        return inmask, n_inside

802        self,

804        lons: np.ndarray,
