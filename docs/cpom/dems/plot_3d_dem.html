  64def _open_dem_store(path: str):
  65    root = zarr.open(path, mode="r")
  66    if isinstance(root, zarr.Array):
  67        arr = root
  68    else:
  69        arr = None
  70        for _, obj in root.arrays():
  71            if obj.ndim == 2:
  72                arr = obj
  73                break
  74        if arr is None:
  75            raise ValueError("No 2D array found in Zarr group.")
  76    attrs = getattr(arr, "attrs", {}) or {}
  77    ncols = int(attrs.get("ncols", arr.shape[1]))
  78    nrows = int(attrs.get("nrows", arr.shape[0]))
  79    binsize = float(attrs.get("binsize", 1000.0))
  80    void_value = attrs.get("void_value", None)
  81    crs_str = attrs.get("crs", "EPSG:3031")
  82    if 'NSIDC Sea Ice Polar Stereographic North' in crs_str:
  83        crs_str="EPSG:3413"
  84        crs_epsg=3413
  85    else:
  86        crs_epsg = int(crs_str.split(":")[1]) if ":" in crs_str else 3031
  87    tf = attrs.get("transform", None)
  88    if tf is None:
  89        top_l = attrs.get("top_l", None)
  90        if top_l is None:
  91            raise ValueError("Missing affine transform in attrs and no fallback.")
  92        x0, y0 = float(top_l[0]), float(top_l[1])
  93        a = binsize; e = -binsize
  94        tf = (x0, a, 0.0, y0, 0.0, e)
  95    else:
  96        tf = tuple(tf)
  97    meta = DemMeta(tf, nrows, ncols, binsize, void_value, crs_epsg)
  98    return arr, meta

 100def _project_lonlat_to_epsg(lon: float, lat: float, crs_epsg:int):
 101    transformer = Transformer.from_crs(CRS.from_epsg(4326), CRS.from_epsg(crs_epsg), always_xy=True)
 102    return transformer.transform(lon, lat)

 104def _xy_to_rowcol(x: float, y: float, tf: Tuple[float, float, float, float, float, float]):
 105    x0, a, b, y0, d, e = tf
 106    if abs(b) > 1e-9 or abs(d) > 1e-9:
 107        det = a*e - b*d
 108        if abs(det) < 1e-12:
 109            raise ValueError("Non-invertible affine transform.")
 110        col = (e*(x-x0) - b*(y-y0))/det
 111        row = (-d*(x-x0) + a*(y-y0))/det
 112    else:
 113        col = (x-x0)/a
 114        row = (y-y0)/e
 115    return float(row), float(col)

 117def _window_indices(center_row: float, center_col: float,
 118                    height_km: float, width_km: float, meta: DemMeta):
 119    """Compute row/col slices that may extend beyond DEM boundaries."""
 120    px_per_km = 1000.0 / meta.binsize
 121    half_rows = (height_km * px_per_km) / 2.0
 122    half_cols = (width_km * px_per_km) / 2.0
 123
 124    r0 = int(np.floor(center_row - half_rows))
 125    r1 = int(np.ceil(center_row + half_rows))
 126    c0 = int(np.floor(center_col - half_cols))
 127    c1 = int(np.ceil(center_col + half_cols))
 128
 129    # NOTE: don’t clamp here — allow negatives and overshoot
 130    return slice(r0, r1), slice(c0, c1)

 133def _read_dem_window(
 134    arr: zarr.Array,
 135    rs: slice,
 136    cs: slice,
 137    void_value: Optional[Union[int, float]],
 138    sea: bool = False,
 139    sea_value: float = 0.0,
 140) -> np.ndarray:
 141    """
 142    Extract DEM data for the requested window.
 143
 144    - Inside DEM extent → real data
 145    - Outside DEM extent →
 146        * NaN if sea=True (so plotting code shows sea)
 147        * sea_value otherwise
 148    """
 149    nrows, ncols = arr.shape
 150
 151    # Requested output shape
 152    out_h = rs.stop - rs.start
 153    out_w = cs.stop - cs.start
 154
 155    # Fill outside DEM
 156    fill_val = np.nan if sea else sea_value
 157    data = np.full((out_h, out_w), fill_val, dtype=np.float32)
 158
 159    # Overlap with DEM
 160    r0c = max(0, rs.start); r1c = min(nrows, rs.stop)
 161    c0c = max(0, cs.start); c1c = min(ncols, cs.stop)
 162
 163    if r0c < r1c and c0c < c1c:
 164        sub = arr[r0c:r1c, c0c:c1c].astype(np.float32)
 165        if void_value is not None:
 166            sub[sub == void_value] = fill_val
 167        data[(r0c - rs.start):(r1c - rs.start),
 168             (c0c - cs.start):(c1c - cs.start)] = sub
 169
 170    return data

 174# ----------------- Shading & resampling -----------------
 175def _hillshade(elev: np.ndarray, dx: float, dy: float,
 176               azimuth: float = 315.0, altitude: float = 45.0,
 177               contrast: float = 1.5, vert_exag: float = 1.0) -> np.ndarray:
 178    az = np.deg2rad(360 - azimuth + 90)
 179    alt = np.deg2rad(altitude)
 180    gy, gx = np.gradient(elev*vert_exag, dy, dx)
 181    slope = np.arctan(np.hypot(gx, gy))
 182    aspect = np.arctan2(-gx, gy)
 183    hs = (np.sin(alt)*np.cos(slope) + np.cos(alt)*np.sin(slope)*np.cos(az - aspect))
 184    lo, hi = np.nanpercentile(hs, (2, 98))
 185    hs = np.clip((hs - lo)/(hi - lo + 1e-9), 0, 1)
 186    hs = hs ** (1.0/max(contrast, 1e-6))
 187    return hs

 198def _deg_label(lat_or_lon: float, is_lat: bool) -> str:
 199    if is_lat:

 206def _plot_graticules(ax, extent_xy,
 207                     lat_step=10, lon_step=20, lat_min=-89.5, lat_max=-50.0,
 208                     z_level=5.0, color="k", lw=0.8, alpha=0.8, crs_epsg=3031,
 209                     ocean_interp: Optional[RegularGridInterpolator] = None,
 210                     draw_labels: bool = False, label_size: float = 9.0,
 211                     label_color: Optional[str] = None, label_offset_km: float = 40.0):
 212    from pyproj import Transformer
 213    xmin, xmax, ymin, ymax = extent_xy
 214    tf = Transformer.from_crs(4326, crs_epsg, always_xy=True)
 215    label_color = label_color or color
 216    xoff = label_offset_km * 1000.0
 217    yoff = label_offset_km * 1000.0
 218
 219    def _clip(xs, ys):
 220        m = (xs >= xmin) & (xs <= xmax) & (ys >= ymin) & (ys <= ymax)
 221        return np.where(m, xs, np.nan), np.where(m, ys, np.nan)
 222
 223    def _mask(xs, ys):
 224        if ocean_interp is None:
 225            return xs, ys
 226        vals = ocean_interp(np.column_stack([ys, xs]))
 227        ocean = vals > 0.5
 228        return np.where(ocean, xs, np.nan), np.where(ocean, ys, np.nan)
 229
 230    # Lat circles
 231    for lat in np.arange(lat_min, lat_max + 1e-9, lat_step):
 232        lons = np.linspace(-180, 180, 721)
 233        xs, ys = tf.transform(lons, np.full_like(lons, lat))
 234        xs, ys = _clip(xs, ys); xs, ys = _mask(xs, ys)
 235        ax.plot3D(xs, ys, zs=z_level, zdir="z", color=color, lw=lw, alpha=alpha)
 236        if draw_labels:
 237            ax.text(np.nanmax(xs), np.nanmean(ys)+yoff, z_level+10,
 238                    _deg_label(lat, True), color=label_color, fontsize=label_size,
 239                    ha="left", va="center")
 240
 241    # Lon spokes
 242    for lon in np.arange(-180, 180 + 1e-9, lon_step):
 243        lats = np.linspace(lat_min, lat_max, 721)
 244        xs, ys = tf.transform(np.full_like(lats, lon), lats)
 245        xs, ys = _clip(xs, ys); xs, ys = _mask(xs, ys)
 246        ax.plot3D(xs, ys, zs=z_level, zdir="z", color=color, lw=lw, alpha=alpha)
 247        if draw_labels:
 248            ax.text(np.nanmean(xs)+xoff, np.nanmax(ys), z_level+10,
 249                    _deg_label(lon, False), color=label_color, fontsize=label_size,
 250                    ha="center", va="bottom")

 825def find_nc_files(
 826    directory: str, recursive: bool, max_files: int | None, include_string: str | None
 827) -> List[str]:
 828    """
 829    Find .nc or .NC files in the specified directory.
 830
 831    Args:
 832        directory (str): The directory to search for .nc files.
 833        recursive (bool): If True, search recursively in subdirectories.
 834        max_files (int|None): if not None, limit number of files read to this number
 835
 836    Returns:
 837        List[str]: A list of found .nc or .NC files with their full paths.
 838    """
 839    files = []
 840    if recursive:
 841        for root, _, _ in os.walk(directory):
 842            files.extend(glob.glob(os.path.join(root, "*.nc")))
 843            files.extend(glob.glob(os.path.join(root, "*.NC")))
 844    else:
 845        files = glob.glob(os.path.join(directory, "*.nc"))
 846        if len(files) < 1:
 847            files = glob.glob(os.path.join(directory, "*.NC"))
 848
 849    if max_files is not None:
 850        if len(files) > max_files:
 851            files = files[:max_files]
 852
 853    if include_string:
 854        files = [file for file in files if include_string in os.path.basename(file)]
 855
 856    return files

1064def hillshade_pyvista(Z, dx, dy, azimuth=315.0, altitude=45.0,
1065                      contrast=1.5, vert_exag=1.0):
1066    """
1067    Compute hillshade intensity in [0,1] for PyVista DEM rendering.
1068
1069    Parameters
1070    ----------
1071    Z : 2D array
1072        Elevation grid (already windowed).
1073    dx, dy : float
1074        Grid spacing in x and y directions.
1075    azimuth : float
1076        Sun azimuth in degrees (default 315).
1077    altitude : float
1078        Sun altitude in degrees above horizon (default 45).
1079    contrast : float
1080        Contrast adjustment (>1 = higher contrast).
1081    vert_exag : float
1082        Vertical exaggeration applied to slope.
1083
1084    Returns
1085    -------
1086    hs : 2D array
1087        Hillshade intensities scaled to [0, 1].
1088    """
1089    az = np.deg2rad(360 - azimuth + 90)
1090    alt = np.deg2rad(altitude)
1091
1092    # gradients with exaggeration
1093    gy, gx = np.gradient(Z * vert_exag, dy, dx)
1094    slope = np.arctan(np.hypot(gx, gy))
1095    aspect = np.arctan2(-gx, gy)
1096
1097    # hillshade formula
1098    hs = (np.sin(alt) * np.cos(slope) +
1099          np.cos(alt) * np.sin(slope) * np.cos(az - aspect))
1100
1101    # normalize to 0–1 with contrast stretch
1102    lo, hi = np.nanpercentile(hs, (2, 98))
1103    hs = np.clip((hs - lo) / (hi - lo + 1e-9), 0, 1)
1104    hs = hs ** (1.0 / max(contrast, 1e-6))
1105
1106    return hs

1066                      contrast=1.5, vert_exag=1.0):
1067    """
1068    Compute hillshade intensity in [0,1] for PyVista DEM rendering.
1069
1070    Parameters
1071    ----------
1072    Z : 2D array
1073        Elevation grid (already windowed).
1074    dx, dy : float
1075        Grid spacing in x and y directions.
1076    azimuth : float
1077        Sun azimuth in degrees (default 315).
1078    altitude : float
1079        Sun altitude in degrees above horizon (default 45).
1080    contrast : float

1082    vert_exag : float
1083        Vertical exaggeration applied to slope.
1084
1085    Returns
1086    -------
1087    hs : 2D array

1089    """

1091    alt = np.deg2rad(altitude)
1092
1093    # gradients with exaggeration
1094    gy, gx = np.gradient(Z * vert_exag, dy, dx)
1095    slope = np.arctan(np.hypot(gx, gy))
1096    aspect = np.arctan2(-gx, gy)
1097
1098    # hillshade formula
1099    hs = (np.sin(alt) * np.cos(slope) +
1100          np.cos(alt) * np.sin(slope) * np.cos(az - aspect))
1101
1102    # normalize to 0–1 with contrast stretch
1103    lo, hi = np.nanpercentile(hs, (2, 98))
1104    hs = np.clip((hs - lo) / (hi - lo + 1e-9), 0, 1)
1105    hs = hs ** (1.0 / max(contrast, 1e-6))
1106
1107    return hs
